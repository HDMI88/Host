chance=95

-- risk factor. Less is riskier!
fator=1000000
-- 1% loss
floss=0.99
-- 2% Gain
stopgain=balance * 1.02

-- streak after which reduction is applied
smax=40
fator2=fator * 1.5


multiplier=5
vquedas=0
maxquedas=2

bal={}
bal[0]=balance
base=balance/fator
if(base<0.0000002) then
   base=0.0000002
end

stoploss=balance*floss
stopprofit=false
nextbet=base
resetseed()
resetstats()
sorte=0
stopwin=false
stopprofit=false
vfator=fator

function dobet()
   highlow()

   status()
   if win then
      if(currentstreak>smax) then
         vfator=fator2
      else
         vfator=fator
      end
      if (stopwin==true) then
         stop()
      end
      if(balance>bal[0]) then
         if (stopprofit==true) then
            stop()
         end
         testasaida()
         bal[0]=balance
         vquedas=0
         base=balance/vfator
         if(base<0.0000002) then
            base=0.0000002
         end
         nextbet=base
         ajustanextbet()
         stoploss=balance*floss
      else
         nextbet=previousbet
      end
   else
      vquedas +=1

      if(vquedas==maxquedas) then
         vquedas=0
         nextbet=previousbet /2
      else
         nextbet=previousbet * multiplier
         ajustanextbet()
      end

      if((balance-nextbet)<stoploss) then
         print ("StopLoss")
         stop()
      end


   end

   calcsorte()
end

function status()
   print("Target:"..stopgain)
   print("S.Loss:"..stoploss)
   print("Profit :"..string.format("%.8f",profit))
   print("Streak:"..currentstreak)
   print("W/L:"..wins/losses)
   print("--------------------")
   print("X5_Div2.txt")
   print("Fator :"..vfator)
   print("multiplier :"..multiplier)
   print("Loss: "..vquedas)
   print("Luck :"..string.format("%.2f",sorte))
end

function testasaida()
   if(balance>=stopgain) then

   if (stopprofit==true) then
      if(balance>bal[0]) then
         print("Saindo... StopProfit")

      end
   end
end

function calcsorte()
   sorte=((wins/bets)/(chance/100))*100
end

function highlow()
   if(lastBet.roll>chance) then
      bethigh=false
   end
   if(lastBet.roll<(100-chance)) then
      bethigh=true
   end
end

function ajustanextbet()
   minbet  = 0.00000020
   x       = nextbet
   nextbet = (math.floor(x/minbet))*minbet
   if(nextbet<minbet) then
      nextbet = minbet
   end
end
